# تطوير منصة "مهاراتي" - دليل البرمجة الشامل
نظرة عامة على المنصة
"مهاراتي" - منصة تعليم المهارات الحياتية للشباب المصري
المنصة تهدف إلى سد الفجوة الكبيرة في التعليم العملي للمهارات الحياتية في مصر، حيث يتخرج الطلاب من الجامعات بمؤهلات أكاديمية لكن بدون مهارات عملية للحياة والعمل.
التخصص للمجتمع المصري
المحتوى المحلي:

أمثلة وحالات دراسة من البيئة المصرية (مثل كيفية إدارة راتب 5000 جنيه شهرياً)
التعامل مع التحديات المحلية مثل الغلاء، البطالة، والضغوط الاجتماعية
مهارات التفاوض في السوق المصري والثقافة التجارية المحلية
التعامل مع البيروقراطية الحكومية والإجراءات الرسمية

اللغة والثقافة:

محتوى باللغة العربية مع استخدام العامية المصرية عند الحاجة
مراعاة القيم والتقاليد المصرية في تقديم النصائح
أمثلة من الحياة اليومية المصرية (المواصلات، العمل، العلاقات الاجتماعية)

المهارات الأساسية المقترحة
1. إدارة المال:

تخطيط الميزانية الشخصية بالجنيه المصري
الادخار في ظل التضخم
الاستثمار البسيط (شهادات البنوك، الذهب، العملات)
التعامل مع البنوك المصرية وفتح الحسابات

2. مهارات العمل:

كتابة السيرة الذاتية باللغة العربية والإنجليزية
التحضير لمقابلات العمل في الشركات المصرية
مهارات التفاوض على الراتب
آداب العمل في البيئة المصرية

3. التواصل والعلاقات:

التواصل الفعال مع الأهل والأصدقاء
إدارة الخلافات والصراعات
بناء شبكة علاقات مهنية (Networking)
التعامل مع ضغوط الزواج والالتزامات الاجتماعية

4. الصحة النفسية:

إدارة التوتر والقلق
التعامل مع الإحباط من الأوضاع الاقتصادية
بناء الثقة بالنفس
تحديد الأهداف الشخصية والمهنية

التقنيات والأدوات
الذكاء الاصطناعي المخصص:

استبيان تقييم شامل يحدد احتياجات كل مستخدم
خطط تعليمية مخصصة حسب العمر والمستوى التعليمي والظروف الاجتماعية
روبوت محادثة يفهم اللهجة المصرية ويقدم نصائح عملية
تتبع التقدم وتقديم توصيات محدثة

المحتوى التفاعلي:

فيديوهات قصيرة (3-5 دقائق) بأسلوب شبابي جذاب
محاكاة لمواقف حقيقية (مقابلة عمل، فتح حساب بنكي، التفاوض على شقة)
تحديات يومية عملية (مثل وضع ميزانية لأسبوع، أو ممارسة مهارة تواصل)
اختبارات تفاعلية وألعاب تعليمية

نموذج الأعمال
الإيرادات:

اشتراكات شهرية مدعومة (20-50 جنيه شهرياً)
شراكات مع الجامعات والمدارس الثانوية
دورات متخصصة مدفوعة للشركات
إعلانات من البنوك وشركات التأمين والخدمات المالية

التسعير المناسب:

نسخة مجانية محدودة
اشتراك أساسي: 30 جنيه/شهر
اشتراك متقدم: 60 جنيه/شهر (يشمل جلسات استشارة فردية)

خطة التنفيذ
المرحلة الأولى (3-6 شهور):

بناء MVP للمنصة بـ 20 مهارة أساسية
إطلاق الروبوت المحادث باللغة العربية
اختبار المحتوى مع 100 مستخدم تجريبي من جامعات مصرية مختلفة

المرحلة الثانية (6-12 شهر):

إضافة المزيد من المحتوى والمهارات
تطوير تطبيق الهاتف المحمول
الشراكة مع 5 جامعات مصرية كبيرة

المرحلة الثالثة (سنة ثانية):

التوسع لتشمل 50,000 مستخدم
إضافة مهارات متقدمة ومحتوى متخصص
إطلاق برنامج للمدربين المحليين

التحديات والحلول
التحديات:

مقاومة الثقافة التقليدية للتعلم الإلكتروني
محدودية الوصول للإنترنت في بعض المناطق
المنافسة مع منصات عالمية

الحلول:

تقديم محتوى عملي ومفيد فوراً
تصميم المنصة للعمل بسرعات إنترنت منخفضة
التركيز على الخصوصية المحلية والفهم العميق للثقافة المصرية

## 2. هيكل قاعدة البيانات (MySQL Schema)

```sql
-- Database Structure for Maharati Platform

CREATE DATABASE maharati_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- Users Table
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    phone VARCHAR(20),
    age INT,
    governorate VARCHAR(50),
    education_level ENUM('secondary', 'university', 'graduate', 'postgraduate'),
    current_income DECIMAL(10,2) DEFAULT 0.00,
    subscription_type ENUM('free', 'basic', 'premium') DEFAULT 'free',
    subscription_expires DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_login TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    is_active BOOLEAN DEFAULT TRUE
);

-- Skills Categories
CREATE TABLE skill_categories (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name_ar VARCHAR(100) NOT NULL,
    description_ar TEXT,
    icon_class VARCHAR(50),
    color_code VARCHAR(7),
    sort_order INT DEFAULT 0
);

-- Individual Skills
CREATE TABLE skills (
    id INT PRIMARY KEY AUTO_INCREMENT,
    category_id INT,
    title_ar VARCHAR(150) NOT NULL,
    description_ar TEXT,
    content_ar LONGTEXT,
    difficulty_level ENUM('beginner', 'intermediate', 'advanced'),
    estimated_time INT, -- in minutes
    practical_examples TEXT,
    egyptian_context TEXT,
    icon_name VARCHAR(50),
    is_premium BOOLEAN DEFAULT FALSE,
    views_count INT DEFAULT 0,
    completion_rate DECIMAL(5,2) DEFAULT 0.00,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (category_id) REFERENCES skill_categories(id)
);

-- User Progress Tracking
CREATE TABLE user_progress (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    skill_id INT NOT NULL,
    progress_percentage DECIMAL(5,2) DEFAULT 0.00,
    completed_at TIMESTAMP NULL,
    time_spent INT DEFAULT 0, -- in minutes
    notes TEXT,
    rating INT CHECK (rating >= 1 AND rating <= 5),
    last_accessed TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (skill_id) REFERENCES skills(id),
    UNIQUE KEY unique_user_skill (user_id, skill_id)
);

-- Chat Conversations with Gemini
CREATE TABLE chat_conversations (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    session_id VARCHAR(100) NOT NULL,
    message_type ENUM('user', 'bot') NOT NULL,
    message_content TEXT NOT NULL,
    skill_context INT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (skill_context) REFERENCES skills(id)
);

-- Daily Challenges
CREATE TABLE daily_challenges (
    id INT PRIMARY KEY AUTO_INCREMENT,
    title_ar VARCHAR(200) NOT NULL,
    description_ar TEXT NOT NULL,
    challenge_date DATE NOT NULL,
    skill_category_id INT,
    difficulty ENUM('easy', 'medium', 'hard'),
    reward_points INT DEFAULT 10,
    egyptian_example TEXT,
    FOREIGN KEY (skill_category_id) REFERENCES skill_categories(id)
);

-- User Challenge Completions
CREATE TABLE user_challenges (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    challenge_id INT NOT NULL,
    completed_at TIMESTAMP NULL,
    user_response TEXT,
    earned_points INT DEFAULT 0,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (challenge_id) REFERENCES daily_challenges(id),
    UNIQUE KEY unique_user_challenge (user_id, challenge_id)
);
```

## 4. PHP Backend Structure

### Main Configuration (config.php):

```php
<?php
// Database Configuration
define('DB_HOST', 'localhost');
define('DB_NAME', 'maharati_db');
define('DB_USER', 'maharati_user');
define('DB_PASS', 'strong_password_here');
define('DB_CHARSET', 'utf8mb4');

// Gemini AI Configuration
define('GEMINI_API_KEY', 'your_gemini_api_key_here');
define('GEMINI_API_URL', 'https://generativelanguage.googleapis.com/v1beta/models/gemini-pro:generateContent');

// Platform Settings
define('SITE_NAME', 'مهاراتي - منصة المهارات الحياتية');
define('SITE_URL', 'https://maharati.com');
define('DEFAULT_TIMEZONE', 'Africa/Cairo');
define('SUBSCRIPTION_BASIC_PRICE', 30.00);
define('SUBSCRIPTION_PREMIUM_PRICE', 60.00);

// Security Settings
define('ENCRYPTION_KEY', 'your_encryption_key_here');
define('SESSION_TIMEOUT', 3600); // 1 hour

// Set timezone
date_default_timezone_set(DEFAULT_TIMEZONE);

// Start session with secure settings
ini_set('session.cookie_httponly', 1);
ini_set('session.cookie_secure', 1);
ini_set('session.use_strict_mode', 1);
session_start();

// Database Connection Class
class Database {
    private static $instance = null;
    private $pdo;
    
    private function __construct() {
        try {
            $dsn = "mysql:host=" . DB_HOST . ";dbname=" . DB_NAME . ";charset=" . DB_CHARSET;
            $this->pdo = new PDO($dsn, DB_USER, DB_PASS, [
                PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION,
                PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
                PDO::ATTR_EMULATE_PREPARES => false,
                PDO::MYSQL_ATTR_INIT_COMMAND => "SET NAMES utf8mb4 COLLATE utf8mb4_unicode_ci"
            ]);
        } catch(PDOException $e) {
            error_log("Database connection failed: " . $e->getMessage());
            die("خطأ في الاتصال بقاعدة البيانات");
        }
    }
    
    public static function getInstance() {
        if (self::$instance === null) {
            self::$instance = new self();
        }
        return self::$instance;
    }
    
    public function getConnection() {
        return $this->pdo;
    }
}

// Gemini AI Integration Class
class GeminiAI {
    private $apiKey;
    private $apiUrl;
    
    public function __construct() {
        $this->apiKey = GEMINI_API_KEY;
        $this->apiUrl = GEMINI_API_URL;
    }
    
    public function generateResponse($userMessage, $userProfile = [], $conversationHistory = []) {
        $contextualPrompt = $this->buildContextualPrompt($userMessage, $userProfile, $conversationHistory);
        
        $requestData = [
            'contents' => [
                [
                    'parts' => [
                        ['text' => $contextualPrompt]
                    ]
                ]
            ],
            'generationConfig' => [
                'temperature' => 0.7,
                'topK' => 40,
                'topP' => 0.95,
                'maxOutputTokens' => 1024,
            ]
        ];
        
        $ch = curl_init();
        curl_setopt_array($ch, [
            CURLOPT_URL => $this->apiUrl . '?key=' . $this->apiKey,
            CURLOPT_RETURNTRANSFER => true,
            CURLOPT_POST => true,
            CURLOPT_POSTFIELDS => json_encode($requestData, JSON_UNESCAPED_UNICODE),
            CURLOPT_HTTPHEADER => [
                'Content-Type: application/json',
                'Accept: application/json'
            ],
            CURLOPT_TIMEOUT => 30,
            CURLOPT_SSL_VERIFYPEER => true
        ]);
        
        $response = curl_exec($ch);
        $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
        $error = curl_error($ch);
        curl_close($ch);
        
        if ($error) {
            error_log("Gemini API cURL Error: " . $error);
            return "عذراً، حدث خطأ في الاتصال. يرجى المحاولة مرة أخرى.";
        }
        
        if ($httpCode !== 200) {
            error_log("Gemini API HTTP Error: " . $httpCode . " - " . $response);
            return "عذراً، الخدمة غير متاحة حالياً. يرجى المحاولة لاحقاً.";
        }
        
        $decodedResponse = json_decode($response, true);
        
        if (isset($decodedResponse['candidates'][0]['content']['parts'][0]['text'])) {
            return $decodedResponse['candidates'][0]['content']['parts'][0]['text'];
        }
        
        return "عذراً، لم أتمكن من فهم سؤالك. يمكنك إعادة صياغته؟";
    }
    
    private function buildContextualPrompt($userMessage, $userProfile, $conversationHistory) {
        $basePrompt = "أنت 'مهاراتي بوت' - مساعد ذكي متخصص في تعليم المهارات الحياتية للشباب المصري..."; // الـ prompt الأساسي
        
        $contextInfo = "";
        
        // Add user profile context
        if (!empty($userProfile)) {
            $contextInfo .= "\nمعلومات المستخدم:\n";
            if (isset($userProfile['name'])) $contextInfo .= "- الاسم: " . $userProfile['name'] . "\n";
            if (isset($userProfile['age'])) $contextInfo .= "- العمر: " . $userProfile['age'] . " سنة\n";
            if (isset($userProfile['governorate'])) $contextInfo .= "- المحافظة: " . $userProfile['governorate'] . "\n";
            if (isset($userProfile['current_income'])) $contextInfo .= "- الراتب الشهري: " . $userProfile['current_income'] . " جنيه\n";
            if (isset($userProfile['education_level'])) $contextInfo .= "- المستوى التعليمي: " . $userProfile['education_level'] . "\n";
        }
        
        // Add conversation history
        if (!empty($conversationHistory)) {
            $contextInfo .= "\nالمحادثات السابقة:\n";
            foreach (array_slice($conversationHistory, -5) as $msg) { // Last 5 messages
                $contextInfo .= "- " . $msg['message_type'] . ": " . $msg['message_content'] . "\n";
            }
```

### Skills API (api/skills.php):

```php
<?php
require_once '../config.php';

header('Content-Type: application/json; charset=utf-8');

class SkillsAPI {
    private $db;
    
    public function __construct() {
        $this->db = Database::getInstance()->getConnection();
    }
    
    public function handleRequest() {
        $method = $_SERVER['REQUEST_METHOD'];
        $action = $_GET['action'] ?? '';
        
        switch ($method) {
            case 'GET':
                $this->handleGet($action);
                break;
            case 'POST':
                $this->handlePost($action);
                break;
            default:
                $this->sendError(405, 'Method not allowed');
        }
    }
    
    private function handleGet($action) {
        switch ($action) {
            case 'categories':
                $this->getCategories();
                break;
            case 'skills':
                $this->getSkills();
                break;
            case 'skill':
                $this->getSkill();
                break;
            case 'user-progress':
                $this->getUserProgress();
                break;
            case 'daily-challenge':
                $this->getDailyChallenge();
                break;
            default:
                $this->sendError(404, 'Action not found');
        }
    }
    
    private function getCategories() {
        try {
            $stmt = $this->db->query("
                SELECT c.*, COUNT(s.id) as skills_count
                FROM skill_categories c
                LEFT JOIN skills s ON c.id = s.category_id
                WHERE c.id IS NOT NULL
                GROUP BY c.id
                ORDER BY c.sort_order ASC
            ");
            
            $categories = $stmt->fetchAll();
            $this->sendSuccess($categories);
        } catch (Exception $e) {
            $this->sendError(500, 'خطأ في جلب التصنيفات');
        }
    }
    
    private function getSkills() {
        $categoryId = $_GET['category_id'] ?? null;
        $limit = intval($_GET['limit'] ?? 20);
        $offset = intval($_GET['offset'] ?? 0);
        
        try {
            $whereClause = $categoryId ? "WHERE s.category_id = ?" : "";
            $params = $categoryId ? [$categoryId] : [];
            
            $stmt = $this->db->prepare("
                SELECT s.*, c.name_ar as category_name,
                       AVG(up.progress_percentage) as avg_progress,
                       COUNT(up.user_id) as learners_count
                FROM skills s
                LEFT JOIN skill_categories c ON s.category_id = c.id
                LEFT JOIN user_progress up ON s.id = up.skill_id
                $whereClause
                GROUP BY s.id
                ORDER BY s.views_count DESC
                LIMIT ? OFFSET ?
            ");
            
            $params = array_merge($params, [$limit, $offset]);
            $stmt->execute($params);
            
            $skills = $stmt->fetchAll();
            $this->sendSuccess($skills);
        } catch (Exception $e) {
            $this->sendError(500, 'خطأ في جلب المهارات');
        }
    }
    
    private function getSkill() {
        $skillId = $_GET['id'] ?? null;
        
        if (!$skillId) {
            $this->sendError(400, 'معرف المهارة مطلوب');
            return;
        }
        
        try {
            // Get skill details
            $stmt = $this->db->prepare("
                SELECT s.*, c.name_ar as category_name
                FROM skills s
                LEFT JOIN skill_categories c ON s.category_id = c.id
                WHERE s.id = ?
            ");
            $stmt->execute([$skillId]);
            $skill = $stmt->fetch();
            
            if (!$skill) {
                $this->sendError(404, 'المهارة غير موجودة');
                return;
            }
            
            // Update view count
            $stmt = $this->db->prepare("UPDATE skills SET views_count = views_count + 1 WHERE id = ?");
            $stmt->execute([$skillId]);
            
            // Get user progress if logged in
            if (isset($_SESSION['user_id'])) {
                $stmt = $this->db->prepare("
                    SELECT progress_percentage, completed_at, time_spent, rating
                    FROM user_progress 
                    WHERE user_id = ? AND skill_id = ?
                ");
                $stmt->execute([$_SESSION['user_id'], $skillId]);
                $skill['user_progress'] = $stmt->fetch();
            }
            
            $this->sendSuccess($skill);
        } catch (Exception $e) {
            $this->sendError(500, 'خطأ في جلب المهارة');
        }
    }
    
    private function getUserProgress() {
        if (!isset($_SESSION['user_id'])) {
            $this->sendError(401, 'تسجيل الدخول مطلوب');
            return;
        }
        
        try {
            $stmt = $this->db->prepare("
                SELECT s.title_ar, s.icon_name, up.progress_percentage, 
                       up.completed_at, up.time_spent, c.name_ar as category_name
                FROM user_progress up
                JOIN skills s ON up.skill_id = s.id
                JOIN skill_categories c ON s.category_id = c.id
                WHERE up.user_id = ?
                ORDER BY up.last_accessed DESC
            ");
            $stmt->execute([$_SESSION['user_id']]);
            
            $progress = $stmt->fetchAll();
            $this->sendSuccess($progress);
        } catch (Exception $e) {
            $this->sendError(500, 'خطأ في جلب التقدم');
        }
    }
    
    private function getDailyChallenge() {
        $today = date('Y-m-d');
        
        try {
            $stmt = $this->db->prepare("
                SELECT dc.*, c.name_ar as category_name
                FROM daily_challenges dc
                LEFT JOIN skill_categories c ON dc.skill_category_id = c.id
                WHERE dc.challenge_date = ?
            ");
            $stmt->execute([$today]);
            
            $challenge = $stmt->fetch();
            
            if (!$challenge) {
                $this->sendError(404, 'لا يوجد تحدي اليوم');
                return;
            }
            
            // Check if user completed the challenge
            if (isset($_SESSION['user_id'])) {
                $stmt = $this->db->prepare("
                    SELECT completed_at, earned_points
                    FROM user_challenges 
                    WHERE user_id = ? AND challenge_id = ?
                ");
                $stmt->execute([$_SESSION['user_id'], $challenge['id']]);
                $challenge['user_completion'] = $stmt->fetch();
            }
            
            $this->sendSuccess($challenge);
        } catch (Exception $e) {
            $this->sendError(500, 'خطأ في جلب التحدي');
        }
    }
    
    private function handlePost($action) {
        if (!isset($_SESSION['user_id'])) {
            $this->sendError(401, 'تسجيل الدخول مطلوب');
            return;
        }
        
        switch ($action) {
            case 'update-progress':
                $this->updateProgress();
                break;
            case 'complete-challenge':
                $this->completeChallenge();
                break;
            case 'rate-skill':
                $this->rateSkill();
                break;
            default:
                $this->sendError(404, 'Action not found');
        }
    }
    
    private function updateProgress() {
        $input = json_decode(file_get_contents('php://input'), true);
        
        $skillId = $input['skill_id'] ?? null;
        $progress = floatval($input['progress'] ?? 0);
        $timeSpent = intval($input['time_spent'] ?? 0);
        
        if (!$skillId || $progress < 0 || $progress > 100) {
            $this->sendError(400, 'بيانات غير صحيحة');
            return;
        }
        
        try {
            $stmt = $this->db->prepare("
                INSERT INTO user_progress (user_id, skill_id, progress_percentage, time_spent)
                VALUES (?, ?, ?, ?)
                ON DUPLICATE KEY UPDATE 
                    progress_percentage = VALUES(progress_percentage),
                    time_spent = time_spent + VALUES(time_spent),
                    completed_at = CASE WHEN VALUES(progress_percentage) = 100 THEN NOW() ELSE completed_at END,
                    last_accessed = NOW()
            ");
            
            $stmt->execute([$_SESSION['user_id'], $skillId, $progress, $timeSpent]);
            
            $this->sendSuccess(['message' => 'تم تحديث التقدم بنجاح']);
        } catch (Exception $e) {
            $this->sendError(500, 'خطأ في تحديث التقدم');
        }
    }
    
    private function completeChallenge() {
        $input = json_decode(file_get_contents('php://input'), true);
        
        $challengeId = $input['challenge_id'] ?? null;
        $userResponse = $input['response'] ?? '';
        
        if (!$challengeId) {
            $this->sendError(400, 'معرف التحدي مطلوب');
            return;
        }
        
        try {
            // Get challenge details
            $stmt = $this->db->prepare("SELECT reward_points FROM daily_challenges WHERE id = ?");
            $stmt->execute([$challengeId]);
            $challenge = $stmt->fetch();
            
            if (!$challenge) {
                $this->sendError(404, 'التحدي غير موجود');
                return;
            }
            
            // Complete the challenge
            $stmt = $this->db->prepare("
                INSERT INTO user_challenges (user_id, challenge_id, completed_at, user_response, earned_points)
                VALUES (?, ?, NOW(), ?, ?)
                ON DUPLICATE KEY UPDATE 
                    completed_at = NOW(),
                    user_response = VALUES(user_response)
            ");
            
            $stmt->execute([$_SESSION['user_id'], $challengeId, $userResponse, $challenge['reward_points']]);
            
            $this->sendSuccess([
                'message' => 'تم إكمال التحدي بنجاح!',
                'earned_points' => $challenge['reward_points']
            ]);
        } catch (Exception $e) {
            $this->sendError(500, 'خطأ في إكمال التحدي');
        }
    }
    
    private function rateSkill() {
        $input = json_decode(file_get_contents('php://input'), true);
        
        $skillId = $input['skill_id'] ?? null;
        $rating = intval($input['rating'] ?? 0);
        
        if (!$skillId || $rating < 1 || $rating > 5) {
            $this->sendError(400, 'بيانات غير صحيحة');
            return;
        }
        
        try {
            $stmt = $this->db->prepare("
                UPDATE user_progress 
                SET rating = ? 
                WHERE user_id = ? AND skill_id = ?
            ");
            
            $stmt->execute([$rating, $_SESSION['user_id'], $skillId]);
            
            if ($stmt->rowCount() > 0) {
                $this->sendSuccess(['message' => 'تم تقييم المهارة بنجاح']);
            } else {
                $this->sendError(400, 'لا يمكن تقييم مهارة لم تبدأ بتعلمها');
            }
        } catch (Exception $e) {
            $this->sendError(500, 'خطأ في التقييم');
        }
    }
    
    private function sendSuccess($data) {
        echo json_encode([
            'success' => true,
            'data' => $data,
            'timestamp' => date('c')
        ], JSON_UNESCAPED_UNICODE);
    }
    
    private function sendError($code, $message) {
        http_response_code($code);
        echo json_encode([
            'success' => false,
            'error' => $message,
            'timestamp' => date('c')
        ], JSON_UNESCAPED_UNICODE);
    }
}
```
